void setup() {
  u8g2.setColorIndex(1);
}

void loop() {
  prev_item = item_selected - 1;
  if (prev_item < 1) {prev_item = 3;}
  next_item = item_selected + 1;
  if (next_item > 3) {next_item = 1;}
  
  //draw background
  u8g2.clearBuffer();
  u8g2.setFontMode(1);
  u8g2.setBitmapMode(1);
  // TITTLE
  u8g2.setFont(u8g2_font_4x6_tr);
  u8g2.drawStr(5, 9, "Menu");

  // AU
  u8g2.drawXBM(119, 12, 5, 7, image_AU_bits);

  // AD
  u8g2.drawXBM(119, 52, 5, 7, image_AD_bits);

  u8g2.sendBuffer();
}
